function encontrarRimaPorSufixo(palavra) { // Se a palavra for vazia, nula ou undefined, já retorna o fallback imediatamente if (!palavra) { return { palavra: "cara", genero: "f" }; } if (excecoesEspecificas[palavra]) { return excecoesEspecificas[palavra]; } for (const regra of regrasRima) { if (palavra.endsWith(regra.sufixo)) { return regra; } } return { palavra: "cara", genero: "f" }; } function gerarResposta(mensagem) { // Validação inicial para campos vazios if (!mensagem || !mensagem.trim()) { return "A resposta aparecerá aqui conforme você digita..."; } const palavras = mensagem.trim().split(" "); let ultimaPalavra = palavras[palavras.length - 1] .toLowerCase() .normalize("NFD") .replace(/[\u0300-\u036f]/g, "") .replace(/[^\w]/g, ""); // Remove pontuações e símbolos // Se após a limpeza a última palavra sumir (ex: o usuário digitou apenas "!!!"), cancela if (!ultimaPalavra) { return "A resposta aparecerá aqui conforme você digita..."; } const { palavraBase, ehPlural } = normalizarEVerificarPlural(ultimaPalavra); // Agora está protegido contra undefined const item = encontrarRimaPorSufixo(palavraBase || ultimaPalavra); let preposicao = item.genero === "m" ? (ehPlural ? "nos seus" : "no seu") : (ehPlural ? "nas suas" : "na sua"); let substantivoResposta = item.palavra; if (ehPlural) { if (!substantivoResposta.endsWith("s")) { substantivoResposta += substantivoResposta.endsWith("o") || substantivoResposta.endsWith("a") || substantivoResposta.endsWith("e") ? "s" : "es"; } } return `meu pau ${preposicao} ${substantivoResposta}`; }
